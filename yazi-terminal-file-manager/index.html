Custom Themes

# ~/.config/yazi/theme.toml

"$schema" = "https://yazi-rs.github.io/schemas/theme.json"

[manager]
cwd = { fg = "black" }

[mode]
normal_main = { bg = "white", bold = true }
normal_alt  = { fg = "black", bg = "gray" }

[filetype]
rules = [
	# Fallback
	{ name = "*/", fg = "gray" }
]
